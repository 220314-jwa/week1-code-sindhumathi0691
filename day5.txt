</head>                                                                                                                                                                     \\  head  tag
	<body>                                                                                                                                                     //  content of the Html document.
		<h1>Cook your food!</h1>                                                                                                   // header tag h1
		<form action="http://localhost:8080/cook">                                                                        \\  user input  tag in order  to send to a server.
		<form action="http://localhost:8080/cook">                                                                    
			<label for="foodName">Food Name:</label>                                                 // for labelling the input increases usability

			<br> \\ line break(tag)
			<button type="submit">Submit</button>                                                              // "submit" button tag
		</form>                                                                                                                                            \\ closing form tag
	</body>                                                                                                                                                        //closing body tag
</html>                                                                                                                                                                                   //closing  html tag
public static void main(String[] args) {
	Javalin app = Javalin.create();                                                                                                                      // created an object of Javalin
	app.start();                                                                                                                                                    \\  The app is listening for http requests.
	app.get("cook", ctx -> {                                                                                                                              // setting to the path "cook" get handler.
		String foodSent = ctx.formParam("foodName");
		String cookedFood = "cooked " + foodSent;
		ctx.result("Here's your food! <br>" + cookedFood);                                                 // context result
	});
